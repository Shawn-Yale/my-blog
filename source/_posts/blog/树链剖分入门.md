## [P3379【模板】最近公共祖先（LCA）](https://www.luogu.com.cn/problem/P3379)

**LCA 我有三不写：**
- 第一不写暴力，因为树高为 $n$，直接超时
- 第二不写 $tarjan$，因为离线很垃圾
-  第三不写倍增，因为不如树剖
	 —— [幻想家协会会长](https://www.bilibili.com/video/BV1Tx4y1D7rk/?spm_id_from=333.337.search-card.all.click&vd_source=f0489718ccab992000c983a006bde4a5)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
const int N = 5e5 + 10;
vi G[N];
int fa[N], dep[N], top[N], sz[N], son[N];
void dfs1(int u){ // 求重儿子
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    // dep写外面，保证根节点深度为 1
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){ // 组装重边成重链
    top[u] = h;
    if(son[u]) dfs2(son[u], h);
    // 有重儿子的话，先走重儿子，保证重儿子的时间戳连续
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        // 重儿子已经走过了，重儿子和父亲一样，都要 continue
        dfs2(v, v);
    }
}
int lca(int x, int y){
    while(top[x] != top[y]){
        if(dep[top[x]] > dep[top[y]]) x = fa[top[x]];
        else y = fa[top[y]];
    }
    return dep[x] < dep[y] ? x : y;
}

void solve(){
    int n, m, s; cin >> n >> m >> s;
    for(int i = 1; i < n; i++){
        int u, v; cin >> u >> v;
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs1(s), dfs2(s, s);
    while(m--){
        int x, y; cin >> x >> y;
        cout << lca(x, y) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [CF1675D Vertical Paths](https://codeforces.com/contest/1675/problem/D)

$最小路径数 = 叶子节点数 = 重链数$

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define vi vector<int>
#define vii vector<vi>
const int N = 2e5 + 10;
vi G[N];
int fa[N], dep[N], top[N], sz[N], son[N];
void dfs1(int u){ // 求重儿子
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    // dep写外面，保证根节点深度为 1
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){ // 组装重边成重链
    top[u] = h;
    if(son[u]) dfs2(son[u], h);
    // 有重儿子的话，先走重儿子，保证重儿子的时间戳连续
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        // 重儿子已经走过了，重儿子和父亲一样，都要 continue
        dfs2(v, v);
    }
}

void solve(){
    int n; cin >> n;
    for(int i = 1; i <= n; i++){ // 不要暴力memset
        G[i].clear();
        fa[i] = dep[i] = top[i] = sz[i] = son[i] = 0;
    }
    int rt;
    for(int i = 1; i <= n; i++){
        int u; cin >> u;
        if(u == i) rt = i;
        else{
            G[u].push_back(i);
            G[i].push_back(u);
        }
    }
    dfs1(rt), dfs2(rt, rt);
    vii chains;
    for(int i = 1; i <= n; i++){
        if(top[i] == i){
            vi path;
            int v = i;
            while(v){
                path.push_back(v);
                v = son[v];
            }
            chains.push_back(path);
        }
    }
    cout << chains.size() << endl;
    for(auto &path : chains){
        cout << path.size() << endl;
        for(auto & x : path)
            cout << x << " \n"[x == path.back()];
    }
    cout << endl;
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    cin >> T;
    while(T--) solve();
    return 0;
}
```
