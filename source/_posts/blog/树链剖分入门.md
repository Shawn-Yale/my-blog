## [P3379【模板】最近公共祖先（LCA）](https://www.luogu.com.cn/problem/P3379)

**LCA 我有三不写：**
- 第一不写暴力，因为树高为 $n$，直接超时
- 第二不写 $tarjan$，因为离线很垃圾
-  第三不写倍增，因为不如树剖
	 —— [幻想家协会会长](https://www.bilibili.com/video/BV1Tx4y1D7rk/?spm_id_from=333.337.search-card.all.click&vd_source=f0489718ccab992000c983a006bde4a5)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
const int N = 5e5 + 10;
vi G[N];
int fa[N], dep[N], top[N], sz[N], son[N];
void dfs1(int u){ // 求重儿子
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    // dep写外面，保证根节点深度为 1
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){ // 组装重边成重链
    top[u] = h;
    if(son[u]) dfs2(son[u], h);
    // 有重儿子的话，先走重儿子，保证重儿子的时间戳连续
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        // 重儿子已经走过了，重儿子和父亲一样，都要 continue
        dfs2(v, v);
    }
}
int lca(int x, int y){
    while(top[x] != top[y]){
        if(dep[top[x]] > dep[top[y]]) x = fa[top[x]];
        else y = fa[top[y]];
    }
    return dep[x] < dep[y] ? x : y;
}

void solve(){
    int n, m, s; cin >> n >> m >> s;
    for(int i = 1; i < n; i++){
        int u, v; cin >> u >> v;
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs1(s), dfs2(s, s);
    while(m--){
        int x, y; cin >> x >> y;
        cout << lca(x, y) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [CF1675D Vertical Paths](https://codeforces.com/contest/1675/problem/D)

$最小路径数 = 叶子节点数 = 重链数$

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define vi vector<int>
#define vii vector<vi>
const int N = 2e5 + 10;
vi G[N];
int fa[N], dep[N], top[N], sz[N], son[N];
void dfs1(int u){ // 求重儿子
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    // dep写外面，保证根节点深度为 1
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){ // 组装重边成重链
    top[u] = h;
    if(son[u]) dfs2(son[u], h);
    // 有重儿子的话，先走重儿子，保证重儿子的时间戳连续
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        // 重儿子已经走过了，重儿子和父亲一样，都要 continue
        dfs2(v, v);
    }
}

void solve(){
    int n; cin >> n;
    for(int i = 1; i <= n; i++){ // 不要暴力memset
        G[i].clear();
        fa[i] = dep[i] = top[i] = sz[i] = son[i] = 0;
    }
    int rt;
    for(int i = 1; i <= n; i++){
        int u; cin >> u;
        if(u == i) rt = i;
        else{
            G[u].push_back(i);
            G[i].push_back(u);
        }
    }
    dfs1(rt), dfs2(rt, rt);
    vii chains;
    for(int i = 1; i <= n; i++){
        if(top[i] == i){
            vi path;
            int v = i;
            while(v){
                path.push_back(v);
                v = son[v];
            }
            chains.push_back(path);
        }
    }
    cout << chains.size() << endl;
    for(auto &path : chains){
        cout << path.size() << endl;
        for(auto & x : path)
            cout << x << " \n"[x == path.back()];
    }
    cout << endl;
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P3384【模板】重链剖分/树链剖分](https://www.luogu.com.cn/problem/P3384)

[董晓算法视频讲解 —— b站](https://www.bilibili.com/video/BV18S4y1e7qb?spm_id_from=333.788.videopod.sections&vd_source=f0489718ccab992000c983a006bde4a5)

超绝3.29KB代码长度
```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define vii vector<vi>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 2e5 + 10;
vi G[N];
int w[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;

struct Node{
    int l, r, sum, add;
}tr[N << 2];

void pushup(int u){
    rt.sum = ls.sum + rs.sum;
}
void pushdown(int u){
    if(rt.add){
        ls.sum += (ls.r - ls.l + 1) * rt.add;
        rs.sum += (rs.r - rs.l + 1) * rt.add;
        ls.add += rt.add;
        rs.add += rt.add;
        rt.add = 0;
    }
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], 0};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify(int u, int l, int r, int k){
    if(l <= rt.l && rt.r <= r){
        rt.sum += (rt.r - rt.l + 1) * k;
        rt.add += k;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) modify(u << 1, l, r, k);
    if(r > mid) modify(u << 1 | 1, l, r, k);
    pushup(u);
}
int query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt.sum;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int sum = 0;
    if(l <= mid) sum = query(u << 1, l, r);
    if(r > mid) sum += query(u << 1 | 1, l, r);
    return sum;
}

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = w[u];
    if(!son[u]) return;
    dfs2(son[u], h);
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void modify_path(int u, int v, int k){
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        modify(1, id[top[u]], id[u], k);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    modify(1, id[v], id[u], k);
}
int query_path(int u, int v){
    int ans = 0;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans += query(1, id[top[u]], id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans += query(1, id[v], id[u]);
    return ans;
}
void modify_tree(int u, int k){
    modify(1, id[u], id[u] + sz[u] - 1, k);
}
int query_tree(int u){
    return query(1, id[u], id[u] + sz[u] - 1);
}

void solve(){
    int n, m, r, p; cin >> n >> m >> r >> p;
    for(int i = 1; i <= n; i++) cin >> w[i];
    for(int i = 1; i < n; i++){
        int x, y; cin >> x >> y;
        G[x].push_back(y), G[y].push_back(x);
    }
    dfs1(r), dfs2(r, r), build(1, 1, n);
    for(int i = 1; i <= m; i++){
        int op, x, y, z; cin >> op;
        if(op == 1){
            cin >> x >> y >> z;
            modify_path(x, y, z % p);
        }else if(op == 2){
            cin >> x >> y;
            cout << query_path(x, y) % p << endl;
        }else if(op == 3){
            cin >> x >> z;
            modify_tree(x, z % p);
        }else{
            cin >> x;
            cout << query_tree(x) % p << endl;
        }
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P2590 [ZJOI2008] 树的统计](https://www.luogu.com.cn/problem/P2590)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 3e4 + 10, inf = 0x3f3f3f3f;
vi G[N];
int w[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;
struct Node{
    int l, r, sum, mx;
}tr[N << 2];

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(int &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = w[u];
    if(!son[u]) return;
    dfs2(son[u], h);
    for(int &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.sum = ls.sum + rs.sum;
    rt.mx = max(ls.mx, rs.mx);
}
void pushup(Node &u, Node &l, Node &r){
    u.sum = l.sum + r.sum;
    u.mx = max(l.mx, r.mx);
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], nw[l]};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify(int u, int x, int k){
    if(rt.l == x && rt.r == x){
        rt.sum = rt.mx = k;
        return;
    }
    int mid = rt.l + rt.r >> 1;
    if(x <= mid) modify(u << 1, x, k);
    else modify(u << 1 | 1, x, k);
    pushup(u);
}
Node query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt;
    int mid = rt.l + rt.r >> 1;
    if(r <= mid) return query(u << 1, l, r);
    else if(l > mid) return query(u << 1 | 1, l, r);
    else{
        Node ans = {l, r};
        Node left = query(u << 1, l, r);
        Node right = query(u << 1 | 1, l, r);
        pushup(ans, left, right);
        return ans;
    }
}
int path_mx(int u, int v){
    int ans = -inf;
    while(top[u] ^ top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans = max(ans, query(1, id[top[u]], id[u]).mx);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans = max(ans, query(1, id[v], id[u]).mx);
    return ans;
}
int path_sum(int u, int v){
    int ans = 0;
    while(top[u] ^ top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans += query(1, id[top[u]], id[u]).sum;
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans += query(1, id[v], id[u]).sum;
    return ans;
}

void solve(){
    int n, m; cin >> n;
    for(int i = 1; i < n; i++){
        int x, y; cin >> x >> y;
        G[x].push_back(y), G[y].push_back(x);
    }
    for(int i = 1; i <= n; i++) cin >> w[i];
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    cin >> m;
    string s; int u, v;
    for(int i = 1; i <= m; i++){
        cin >> s >> u >> v;
        if(s == "CHANGE") modify(1, id[u], v);
        else if(s == "QMAX") cout << path_mx(u, v) << endl;
        else cout << path_sum(u, v) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P2146 [NOI2015] 软件包管理器](https://www.luogu.com.cn/problem/P2146)

**显然：**
- 每次安装软件包，就把 **从根节点到该软件包** 路径上的值全部变为 $1$
- 每次卸载软件包，就把 **该软件包以及它的子树** 的值变为 $0$

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 1e5 + 10, inf = 0x3f3f3f3f;
vi G[N];
int w[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;
struct Node{
    int l, r, sum, tag;
    // tag = -1 表示不用修改
    // tag = 0/1 表示区间置 0/1
}tr[N << 2];

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(int &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = w[u];
    if(son[u]) dfs2(son[u], h);
    for(int &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.sum = ls.sum + rs.sum;
}
void pushdown(int u){
    if(~rt.tag){
        ls.sum = (ls.r - ls.l + 1) * rt.tag;
        rs.sum = (rs.r - rs.l + 1) * rt.tag;
        ls.tag = rs.tag = rt.tag;
        rt.tag = -1;
    }
}
void build(int u, int l, int r){
    rt = {l, r, 0, -1};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
}
void modify(int u, int l, int r, int k){
    if(l <= rt.l && rt.r <= r){
        rt.sum = (rt.r - rt.l + 1) * k;
        rt.tag = k;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) modify(u << 1, l, r, k);
    if(r > mid) modify(u << 1 | 1, l, r, k);
    pushup(u);
}
void modify_path(int u, int v, int k){
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        modify(1, id[top[u]], id[u], k);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u , v);
    modify(1, id[v], id[u], k);
}
void modify_tree(int u, int k){
    modify(1, id[u], id[u] + sz[u] - 1, k);
}
int query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt.sum;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int ans = 0;
    if(l <= mid) ans = query(u << 1, l, r);
    if(r > mid) ans += query(u << 1 | 1, l, r);
    return ans;
}
int query_path(int u, int v){
    int ans = 0;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans += query(1, id[top[u]], id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans += query(1, id[v], id[u]);
    return ans;
}
int query_tree(int u){
    return query(1, id[u], id[u] + sz[u] - 1);
}

void solve(){
    int n; cin >> n;
    for(int i = 1; i < n; i++){
        int x; cin >> x;
        G[i + 1].push_back(x + 1), G[x + 1].push_back(i + 1);
    }
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    int m; cin >> m;
    string op; int u;
    for(int i = 1; i <= m; i++){
        cin >> op >> u;
        u++;
        if(op == "install"){
            cout << dep[u] - query_path(u, 1) << endl;
            modify_path(u, 1, 1);
        }else{
            cout << query_tree(u) << endl;
            modify_tree(u, 0);
        }
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P4315 月下“毛景树”](https://www.luogu.com.cn/problem/P4315)(边权转点权模板)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 1e5 + 10;
struct edge{
    int v, w;
};
vector<edge> e[N];

struct Node{
    int l, r, mx, add, tag;
    // tag 表示要变成的数, -1 代表不用变
    // add 表示要加的数
}tr[N << 2];

int val[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;
int eu[N], ev[N];

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(auto &[v, w] : e[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        val[v] = w;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = val[u];
    if(son[u]) dfs2(son[u], h);
    for(auto &[v, w] : e[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.mx = max(ls.mx, rs.mx);
}
void pushdown(int u){
    if(~rt.tag){
        ls.mx = rs.mx = ls.tag = rs.tag = rt.tag;
        ls.add = rs.add = 0;
        rt.tag = -1;
    }
    if(rt.add){
        ls.mx += rt.add;
        rs.mx += rt.add;
        ls.add += rt.add;
        rs.add += rt.add;
        rt.add = 0;
    }
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], 0, -1};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify(int u, int l, int r, int k){
    if(l <= rt.l && rt.r <= r){
        rt.mx = rt.tag = k;
        rt.add = 0;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) modify(u << 1, l, r, k);
    if(r > mid) modify(u << 1 | 1, l, r, k);
    pushup(u);
}
void change(int k, int w){
    int u = eu[k], v = ev[k];
    if(dep[u] < dep[v]) swap(u, v);
    // 边权转点权时，深度大的点存边权
    modify(1, id[u], id[u], w);
}
void cover(int u, int v, int w){
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        modify(1, id[top[u]], id[u], w);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u]) // 排除 LCA 自己，因为边权存在于子节点
        modify(1, id[v] + 1, id[u], w);
}
void add1(int u, int l, int r, int k){
    if(l <= rt.l && rt.r <= r){
        rt.mx += k;
        rt.add += k;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) add1(u << 1, l, r, k);
    if(r > mid) add1(u << 1 | 1, l, r, k);
    pushup(u);
}
void add(int u, int v, int w){
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        add1(1, id[top[u]], id[u], w);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u]) // 排除 LCA 自己，因为边权存在于子节点
        add1(1, id[v] + 1, id[u], w);
}
int query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt.mx;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int mx = 0;
    if(l <= mid) mx = query(u << 1, l, r);
    if(r > mid) mx = max(mx, query(u << 1 | 1, l, r));
    return mx;
}
int query_max(int u, int v){
    int mx = 0;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        mx = max(mx, query(1, id[top[u]], id[u]));
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u]) // 排除 LCA 自己，因为边权存在于子节点
        mx = max(mx, query(1, id[v] + 1, id[u]));
    return mx;
}


void solve(){
    int n; cin >> n;
    for(int i = 1; i < n; i++){
        int u, v, w; cin >> u >> v >> w;
        eu[i] = u, ev[i] = v;
        e[u].push_back({v, w});
        e[v].push_back({u, w});
    }
    fa[1] = 0; dep[1] = 1; val[1] = 0;
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    string s; int k, u, v, w;
    while(cin >> s, s != "Stop"){
        if(s == "Change") cin >> k >> w, change(k, w);
        else if(s == "Cover") cin >> u >> v >> w, cover(u, v, w);
        else if(s == "Add") cin >> u >> v >> w, add(u, v, w);
        else cin >> u >> v, cout << query_max(u, v) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P1505 [国家集训队] 旅游](https://www.luogu.com.cn/problem/P1505)(边权转点权)(码量稍大)

与前四题类似，五倍经验

取反可以用 $*=1$
查询时可以只写 $2$ 个 $query$，同时查询 $max, \ min, \ sum$，返回结构体，没必要像我一样写 $6$ 个 $query$

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 2e5 + 10, inf = 0x3f3f3f3f;
struct edge{
    int v, w;
};
vector<edge> e[N];
int eu[N], ev[N];

struct Node{
    int l, r, mx, mn, sum, tag;
    // tag 标记是否取反
}tr[N << 2];

int val[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(auto &[v, w] : e[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        val[v] = w;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = val[u];
    if(son[u]) dfs2(son[u], h);
    for(auto &[v, w] : e[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.mx = max(ls.mx, rs.mx);
    rt.mn = min(ls.mn, rs.mn);
    rt.sum = ls.sum + rs.sum;
}
void pushdown(int u){
    if(rt.tag){
        swap(ls.mx, ls.mn);
        ls.mx = -ls.mx;
        ls.mn = -ls.mn;
        swap(rs.mx, rs.mn);
        rs.mx = -rs.mx;
        rs.mn = -rs.mn;

        ls.sum = -ls.sum;
        rs.sum = -rs.sum;
        ls.tag = 1 - ls.tag;
        rs.tag = 1 - rs.tag;
        rt.tag = 0;
    }
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], nw[l], nw[l], 0};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify_c(int u, int x, int k){ // 线段树上单点x的值修改成k
    if(rt.l == x && rt.r == x){
        rt.mx = rt.mn = rt.sum = k;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(x <= mid) modify_c(u << 1, x, k);
    else modify_c(u << 1 | 1, x, k);
    pushup(u);
}
void modify_c(int i, int w){ // 将输入的第 i 条边权值改为 w
    int u = eu[i], v = ev[i];
    if(dep[u] < dep[v]) swap(u, v);
    modify_c(1, id[u], w);
}
void modify(int u, int l, int r){ // 线段树上区间取反
    if(l <= rt.l && rt.r <= r){
        swap(rt.mx, rt.mn);
        rt.mx = -rt.mx;
        rt.mn = -rt.mn;
        rt.sum = -rt.sum;
        rt.tag = 1 - rt.tag;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) modify(u << 1, l, r);
    if(r > mid) modify(u << 1 | 1, l, r);
    pushup(u);
}
void modify(int u, int v){ // 将 u, v 节点之间的边权都变为相反数
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        modify(1, id[top[u]], id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u])
        modify(1, id[v] + 1, id[u]);
}
int query_sum(int u, int l, int r){ // 线段树上求区间和
    if(l <= rt.l && rt.r <= r) return rt.sum;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int sum = 0;
    if(l <= mid) sum = query_sum(u << 1, l, r);
    if(r > mid) sum += query_sum(u << 1 | 1, l, r);
    return sum;
}
int query_sum(int u, int v){ // 求 u, v 节点之间的边权和
    int sum = 0;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        sum += query_sum(1, id[top[u]], id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u])
        sum += query_sum(1, id[v] + 1, id[u]);
    return sum;
}
int query_mx(int u, int l, int r){ // 线段树上求区间最大值
    if(l <= rt.l && rt.r <= r) return rt.mx;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int mx = -inf;
    if(l <= mid) mx = query_mx(u << 1, l, r);
    if(r > mid) mx = max(mx, query_mx(u << 1 | 1, l, r));
    return mx;
}
int query_mx(int u, int v){ // 求 u, v 节点之间的边权最大值
    int mx = -inf;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        mx = max(mx, query_mx(1, id[top[u]], id[u]));
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u])
        mx = max(mx, query_mx(1, id[v] + 1, id[u]));
    return mx;
}
int query_mn(int u, int l, int r){ // 线段树上求区间最大值
    if(l <= rt.l && rt.r <= r) return rt.mn;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int mn = inf;
    if(l <= mid) mn = query_mn(u << 1, l, r);
    if(r > mid) mn = min(mn, query_mn(u << 1 | 1, l, r));
    return mn;
}
int query_mn(int u, int v){ // 求 u, v 节点之间的边权最大值
    int mn = inf;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        mn = min(mn, query_mn(1, id[top[u]], id[u]));
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u])
        mn = min(mn, query_mn(1, id[v] + 1, id[u]));
    return mn;
}
void solve(){
    int n; cin >> n;
    for(int i = 1; i < n; i++){
        int u, v, w; cin >> u >> v >> w;
        u++, v++; // 注意题目中给的编号范围为: 0 ~ n-1
        eu[i] = u, ev[i] = v;
        e[u].push_back({v, w});
        e[v].push_back({u, w});
    }
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    int m; cin >> m;
    string s; int u, v;
    for(int i = 1; i <= m; i++){
        cin >> s >> u >> v;
        if(s != "C") u++, v++; // 注意题目中给的编号范围为: 0 ~ n-1
        if(s == "C") modify_c(u, v);
        else if(s == "N") modify(u, v);
        else if(s == "SUM") cout << query_sum(u, v) << endl;
        else if(s == "MAX") cout << query_mx(u, v) << endl;
        else cout << query_mn(u, v) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P4116 Qtree3](https://www.luogu.com.cn/problem/P4116)(思路转化)

我们先看下题，他查询的是 $1→v$ 上的第一个黑点，这种东西该怎么操作呢？

我们发现，从 $1→v$ 路径上的点，他的 $dfs$序是单调递增的，也就是第一个黑点绝对是 $dfs$序最小的。

所以我们考虑给每个黑点赋值为他自己的 $dfs$序的值，每个白点赋值为 $inf$。

如此一来，题目中的查询第一个黑点就变成了查询 $1→v$ 上的最小值。

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 1e5 + 10, inf = 0x3f3f3f3f;
struct Node{
    int l, r, mn;
}tr[N << 2];
vi G[N];

int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;
int qid[N]; // 映射: dfs序 -> 节点编号

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, qid[cnt] = u;
    if(son[u]) dfs2(son[u], h);
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.mn = min(ls.mn, rs.mn);
}
void build(int u, int l, int r){
    rt = {l, r, nw[l]};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify(int u, int x){ // 线段树的下标就是dfs序
    if(rt.l == x && rt.r == x){
        rt.mn = (rt.mn == inf ? x : inf);
        // 如果是白点->黑点，就将其赋值为自身的dfs序
        // 如果是黑点—>白点，就将其赋值为inf
        return;
    }
    int mid = rt.l + rt.r >> 1;
    if(x <= mid) modify(u << 1, x);
    else modify(u << 1 | 1, x);
    pushup(u);
}
int query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt.mn;
    int mid = rt.l + rt.r >> 1;
    int mn = inf;
    if(l <= mid) mn = query(u << 1, l, r);
    if(r > mid) mn = min(mn, query(u << 1 | 1, l, r));
    return mn;
}
int query(int u, int v){
    int mn = inf;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        mn = min(mn, query(1, id[top[u]], id[u]));
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    mn = min(mn, query(1, id[v], id[u]));
    return mn;
}

void solve(){
    int n, m; cin >> n >> m;
    fill(nw + 1, nw + n + 1, inf);
    for(int i = 1; i < n; i++){
        int u, v; cin >> u >> v;
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    int op, x;
    for(int i = 1; i <= m; i++){
        cin >> op >> x;
        if(!op) modify(1, id[x]);
        else{
            int ans = query(1, x);
            cout << (ans == inf ? -1 : qid[ans]) << endl;
        }
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P4180 [BJWC2010] 严格次小生成树](https://www.luogu.com.cn/problem/P4180)(Kruskal算法 + 树剖维护树上区间最大和区间严格次大)

可以证明**次小生成树一定是最小生成树换一条边形成的**，因为换更多的边一定不优

所以先跑出来一个**最小生成树**，然后枚举所有非树边，加入**非树边**之后会形成一个环，然后去掉环上的最大 ( $\neq$ 加入的非树边，严格 $<$ ) 的一条**树边**

但是有一个问题是**如果非树边等于环上的最大树边**那么是**不符合要求**的，所以还要维护一个严格次大的树边。用树剖+线段树维护树上区间最大和严格次大值。

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define pii pair<int, int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 1e5 + 10, M = 3e5 + 10, inf = 0x3f3f3f3f;

int p[N];
struct edge{
    int u, v, w, op; // op标记树边
    bool operator< (const edge &t) const{
        return w < t.w;
    }
}e[M];

int find(int x) {return (p[x] == x ? x : p[x] = find(p[x]));}

struct Edge{
    int v, w;
};
vector<Edge> G[N]; // n 节点的树
int val[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(auto &[v, w] : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        val[v] = w;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = val[u];
    if(son[u]) dfs2(son[u], h);
    for(auto &[v, w] : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}

struct Node{
    int l, r, mx, mx2;
}tr[N << 2];

void pushup(int u){
    rt.mx = max(ls.mx, rs.mx);
    rt.mx2 = max({ls.mx2, rs.mx2, (ls.mx == rs.mx ? -inf : min(ls.mx, rs.mx))});
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], -inf};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void merge(Node &a, const Node &b){
    int mx = max(a.mx, b.mx);
    int mx2 = max({a.mx2, b.mx2, (a.mx == b.mx ? -inf : min(a.mx, b.mx))});
    a = {0, 0, mx, mx2};
}
Node query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt;
    int mid = rt.l + rt.r >> 1;
    Node a = {0, 0, -inf, -inf}, b = {0, 0, -inf, -inf};
    if(l <= mid) a = query(u << 1, l, r);
    if(r > mid) b = query(u << 1 | 1, l, r);
    merge(a, b);
    return a;
}
Node query_tree(int u, int v){
    Node a = {0, 0, -inf, -inf};
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        merge(a, query(1, id[top[u]], id[u]));
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    if(id[v] + 1 <= id[u])
        merge(a, query(1, id[v] + 1, id[u]));
    return a;
}

void solve(){
    int n, m; cin >> n >> m;
    for(int i = 1; i <= m; i++) cin >> e[i].u >> e[i].v >> e[i].w;
    sort(e + 1, e + m + 1);
    iota(p, p + n + 1, 0ll);
    int sum = 0;
    for(int i = 1; i <= m; i++){
        int u = e[i].u, v = e[i].v, w = e[i].w;
        if(find(u) != find(v)){
            p[find(u)] = find(v);
            sum += e[i].w;
            G[u].push_back({v, w});
            G[v].push_back({u, w});
            e[i].op = 1;
        }
    }
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    int dif = inf;
    for(int i = 1; i <= m; i++){
        if(!e[i].op){ // 枚举非树边
            int u = e[i].u, v = e[i].v, w = e[i].w;
            Node a = query_tree(u, v);
            int mx = a.mx, mx2 = a.mx2;
            mx = (mx == w ? mx2 : mx);
            dif = min(dif, w - mx);
        }
    }
    cout << sum + dif << endl;
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```
