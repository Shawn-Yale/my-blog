## [P3379【模板】最近公共祖先（LCA）](https://www.luogu.com.cn/problem/P3379)

**LCA 我有三不写：**
- 第一不写暴力，因为树高为 $n$，直接超时
- 第二不写 $tarjan$，因为离线很垃圾
-  第三不写倍增，因为不如树剖
	 —— [幻想家协会会长](https://www.bilibili.com/video/BV1Tx4y1D7rk/?spm_id_from=333.337.search-card.all.click&vd_source=f0489718ccab992000c983a006bde4a5)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
const int N = 5e5 + 10;
vi G[N];
int fa[N], dep[N], top[N], sz[N], son[N];
void dfs1(int u){ // 求重儿子
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    // dep写外面，保证根节点深度为 1
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){ // 组装重边成重链
    top[u] = h;
    if(son[u]) dfs2(son[u], h);
    // 有重儿子的话，先走重儿子，保证重儿子的时间戳连续
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        // 重儿子已经走过了，重儿子和父亲一样，都要 continue
        dfs2(v, v);
    }
}
int lca(int x, int y){
    while(top[x] != top[y]){
        if(dep[top[x]] > dep[top[y]]) x = fa[top[x]];
        else y = fa[top[y]];
    }
    return dep[x] < dep[y] ? x : y;
}

void solve(){
    int n, m, s; cin >> n >> m >> s;
    for(int i = 1; i < n; i++){
        int u, v; cin >> u >> v;
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs1(s), dfs2(s, s);
    while(m--){
        int x, y; cin >> x >> y;
        cout << lca(x, y) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [CF1675D Vertical Paths](https://codeforces.com/contest/1675/problem/D)

$最小路径数 = 叶子节点数 = 重链数$

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define vi vector<int>
#define vii vector<vi>
const int N = 2e5 + 10;
vi G[N];
int fa[N], dep[N], top[N], sz[N], son[N];
void dfs1(int u){ // 求重儿子
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    // dep写外面，保证根节点深度为 1
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){ // 组装重边成重链
    top[u] = h;
    if(son[u]) dfs2(son[u], h);
    // 有重儿子的话，先走重儿子，保证重儿子的时间戳连续
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        // 重儿子已经走过了，重儿子和父亲一样，都要 continue
        dfs2(v, v);
    }
}

void solve(){
    int n; cin >> n;
    for(int i = 1; i <= n; i++){ // 不要暴力memset
        G[i].clear();
        fa[i] = dep[i] = top[i] = sz[i] = son[i] = 0;
    }
    int rt;
    for(int i = 1; i <= n; i++){
        int u; cin >> u;
        if(u == i) rt = i;
        else{
            G[u].push_back(i);
            G[i].push_back(u);
        }
    }
    dfs1(rt), dfs2(rt, rt);
    vii chains;
    for(int i = 1; i <= n; i++){
        if(top[i] == i){
            vi path;
            int v = i;
            while(v){
                path.push_back(v);
                v = son[v];
            }
            chains.push_back(path);
        }
    }
    cout << chains.size() << endl;
    for(auto &path : chains){
        cout << path.size() << endl;
        for(auto & x : path)
            cout << x << " \n"[x == path.back()];
    }
    cout << endl;
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P3384【模板】重链剖分/树链剖分](https://www.luogu.com.cn/problem/P3384)

[董晓算法视频讲解 —— b站](https://www.bilibili.com/video/BV18S4y1e7qb?spm_id_from=333.788.videopod.sections&vd_source=f0489718ccab992000c983a006bde4a5)

超绝3.29KB代码长度
```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define vii vector<vi>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 2e5 + 10;
vi G[N];
int w[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;

struct Node{
    int l, r, sum, add;
}tr[N << 2];

void pushup(int u){
    rt.sum = ls.sum + rs.sum;
}
void pushdown(int u){
    if(rt.add){
        ls.sum += (ls.r - ls.l + 1) * rt.add;
        rs.sum += (rs.r - rs.l + 1) * rt.add;
        ls.add += rt.add;
        rs.add += rt.add;
        rt.add = 0;
    }
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], 0};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify(int u, int l, int r, int k){
    if(l <= rt.l && rt.r <= r){
        rt.sum += (rt.r - rt.l + 1) * k;
        rt.add += k;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) modify(u << 1, l, r, k);
    if(r > mid) modify(u << 1 | 1, l, r, k);
    pushup(u);
}
int query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt.sum;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int sum = 0;
    if(l <= mid) sum = query(u << 1, l, r);
    if(r > mid) sum += query(u << 1 | 1, l, r);
    return sum;
}

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(auto &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = w[u];
    if(!son[u]) return;
    dfs2(son[u], h);
    for(auto &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void modify_path(int u, int v, int k){
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        modify(1, id[top[u]], id[u], k);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    modify(1, id[v], id[u], k);
}
int query_path(int u, int v){
    int ans = 0;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans += query(1, id[top[u]], id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans += query(1, id[v], id[u]);
    return ans;
}
void modify_tree(int u, int k){
    modify(1, id[u], id[u] + sz[u] - 1, k);
}
int query_tree(int u){
    return query(1, id[u], id[u] + sz[u] - 1);
}

void solve(){
    int n, m, r, p; cin >> n >> m >> r >> p;
    for(int i = 1; i <= n; i++) cin >> w[i];
    for(int i = 1; i < n; i++){
        int x, y; cin >> x >> y;
        G[x].push_back(y), G[y].push_back(x);
    }
    dfs1(r), dfs2(r, r), build(1, 1, n);
    for(int i = 1; i <= m; i++){
        int op, x, y, z; cin >> op;
        if(op == 1){
            cin >> x >> y >> z;
            modify_path(x, y, z % p);
        }else if(op == 2){
            cin >> x >> y;
            cout << query_path(x, y) % p << endl;
        }else if(op == 3){
            cin >> x >> z;
            modify_tree(x, z % p);
        }else{
            cin >> x;
            cout << query_tree(x) % p << endl;
        }
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P2590 [ZJOI2008] 树的统计](https://www.luogu.com.cn/problem/P2590)

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 3e4 + 10, inf = 0x3f3f3f3f;
vi G[N];
int w[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;
struct Node{
    int l, r, sum, mx;
}tr[N << 2];

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(int &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = w[u];
    if(!son[u]) return;
    dfs2(son[u], h);
    for(int &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.sum = ls.sum + rs.sum;
    rt.mx = max(ls.mx, rs.mx);
}
void pushup(Node &u, Node &l, Node &r){
    u.sum = l.sum + r.sum;
    u.mx = max(l.mx, r.mx);
}
void build(int u, int l, int r){
    rt = {l, r, nw[l], nw[l]};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
    pushup(u);
}
void modify(int u, int x, int k){
    if(rt.l == x && rt.r == x){
        rt.sum = rt.mx = k;
        return;
    }
    int mid = rt.l + rt.r >> 1;
    if(x <= mid) modify(u << 1, x, k);
    else modify(u << 1 | 1, x, k);
    pushup(u);
}
Node query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt;
    int mid = rt.l + rt.r >> 1;
    if(r <= mid) return query(u << 1, l, r);
    else if(l > mid) return query(u << 1 | 1, l, r);
    else{
        Node ans = {l, r};
        Node left = query(u << 1, l, r);
        Node right = query(u << 1 | 1, l, r);
        pushup(ans, left, right);
        return ans;
    }
}
int path_mx(int u, int v){
    int ans = -inf;
    while(top[u] ^ top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans = max(ans, query(1, id[top[u]], id[u]).mx);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans = max(ans, query(1, id[v], id[u]).mx);
    return ans;
}
int path_sum(int u, int v){
    int ans = 0;
    while(top[u] ^ top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans += query(1, id[top[u]], id[u]).sum;
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans += query(1, id[v], id[u]).sum;
    return ans;
}

void solve(){
    int n, m; cin >> n;
    for(int i = 1; i < n; i++){
        int x, y; cin >> x >> y;
        G[x].push_back(y), G[y].push_back(x);
    }
    for(int i = 1; i <= n; i++) cin >> w[i];
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    cin >> m;
    string s; int u, v;
    for(int i = 1; i <= m; i++){
        cin >> s >> u >> v;
        if(s == "CHANGE") modify(1, id[u], v);
        else if(s == "QMAX") cout << path_mx(u, v) << endl;
        else cout << path_sum(u, v) << endl;
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```

## [P2146 [NOI2015] 软件包管理器](https://www.luogu.com.cn/problem/P2146)

**显然：**
- 每次安装软件包，就把 **从根节点到该软件包** 路径上的值全部变为 $1$
- 每次卸载软件包，就把 **该软件包以及它的子树** 的值变为 $0$

```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
#define vi vector<int>
#define rt tr[u]
#define ls tr[u << 1]
#define rs tr[u << 1 | 1]
const int N = 1e5 + 10, inf = 0x3f3f3f3f;
vi G[N];
int w[N];
int sz[N], dep[N], fa[N], son[N];
int top[N], id[N], nw[N], cnt;
struct Node{
    int l, r, sum, tag;
    // tag = -1 表示不用修改
    // tag = 0/1 表示区间置 0/1
}tr[N << 2];

void dfs1(int u){
    sz[u] = 1;
    dep[u] = dep[fa[u]] + 1;
    for(int &v : G[u]){
        if(v == fa[u]) continue;
        fa[v] = u;
        dfs1(v);
        sz[u] += sz[v];
        if(sz[v] > sz[son[u]]) son[u] = v;
    }
}
void dfs2(int u, int h){
    top[u] = h, id[u] = ++cnt, nw[cnt] = w[u];
    if(son[u]) dfs2(son[u], h);
    for(int &v : G[u]){
        if(v == fa[u] || v == son[u]) continue;
        dfs2(v, v);
    }
}
void pushup(int u){
    rt.sum = ls.sum + rs.sum;
}
void pushdown(int u){
    if(~rt.tag){
        ls.sum = (ls.r - ls.l + 1) * rt.tag;
        rs.sum = (rs.r - rs.l + 1) * rt.tag;
        ls.tag = rs.tag = rt.tag;
        rt.tag = -1;
    }
}
void build(int u, int l, int r){
    rt = {l, r, 0, -1};
    if(l == r) return;
    int mid = l + r >> 1;
    build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
}
void modify(int u, int l, int r, int k){
    if(l <= rt.l && rt.r <= r){
        rt.sum = (rt.r - rt.l + 1) * k;
        rt.tag = k;
        return;
    }
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    if(l <= mid) modify(u << 1, l, r, k);
    if(r > mid) modify(u << 1 | 1, l, r, k);
    pushup(u);
}
void modify_path(int u, int v, int k){
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        modify(1, id[top[u]], id[u], k);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u , v);
    modify(1, id[v], id[u], k);
}
void modify_tree(int u, int k){
    modify(1, id[u], id[u] + sz[u] - 1, k);
}
int query(int u, int l, int r){
    if(l <= rt.l && rt.r <= r) return rt.sum;
    pushdown(u);
    int mid = rt.l + rt.r >> 1;
    int ans = 0;
    if(l <= mid) ans = query(u << 1, l, r);
    if(r > mid) ans += query(u << 1 | 1, l, r);
    return ans;
}
int query_path(int u, int v){
    int ans = 0;
    while(top[u] != top[v]){
        if(dep[top[u]] < dep[top[v]]) swap(u, v);
        ans += query(1, id[top[u]], id[u]);
        u = fa[top[u]];
    }
    if(dep[u] < dep[v]) swap(u, v);
    ans += query(1, id[v], id[u]);
    return ans;
}
int query_tree(int u){
    return query(1, id[u], id[u] + sz[u] - 1);
}

void solve(){
    int n; cin >> n;
    for(int i = 1; i < n; i++){
        int x; cin >> x;
        G[i + 1].push_back(x + 1), G[x + 1].push_back(i + 1);
    }
    dfs1(1), dfs2(1, 1), build(1, 1, n);
    int m; cin >> m;
    string op; int u;
    for(int i = 1; i <= m; i++){
        cin >> op >> u;
        u++;
        if(op == "install"){
            cout << dep[u] - query_path(u, 1) << endl;
            modify_path(u, 1, 1);
        }else{
            cout << query_tree(u) << endl;
            modify_tree(u, 0);
        }
    }
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    //cin >> T;
    while(T--) solve();
    return 0;
}
```
