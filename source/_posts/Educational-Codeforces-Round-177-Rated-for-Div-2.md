## [CF2086A. Cloudberry Jam](https://codeforces.com/contest/2086/problem/A)
```cpp
#include <bits/stdc++.h>
using namespace std;
#define endl '\n'
#define int long long
int n;

void solve(){
    cin >> n;
    cout << 2 * n << endl;
}

signed main(){
    ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
    int T = 1;
    cin >> T;
    while(T--) solve();
    return 0;
}
```
